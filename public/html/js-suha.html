Fetch API használata Promise-al

const getFetch = (url) => {
  return new Promise((resolve, reject) => {
    fetch(url)
      .then((response) => {
        if (!response.ok) {
          reject(`Hiba: ${response.statusText} (${response.message})`)
        }
        return response.json()
      })
      .then((data) => {
        resolve(data)
      })
      .catch((error) => {
        reject(`Hiba: ${error}`)
      })
  })
}

POST Fetch Promise (Frontend)

Fetch API használata POST metódussal Promise-al

const postFetch = (url, postBody) => {
  return new Promise((resolve, reject) => {
    fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(postBody),
    })
      .then((response) => {
        if (!response.ok) {
          reject(`Hiba: ${response.statusText} (${response.status})`)
        }
        return response.json()
      })
      .then((data) => {
        resolve(data)
      })
      .catch((error) => {
        reject(`Hiba: ${error}`)
      })
  })
}

API GET + paraméter (Backend)

API Get kérés JSON file-ra

app.get('/api/pelda/:parameter', (request, response) => {
  const peldaParam = request.params.parameter
  const filePath = path.join(__dirname, `/public/data/${peldaParam}.json`)
  fs.readFile(filePath, 'utf8', (err, data) => {
    if (err) {
      console.error('Error reading file:', err)
      return response.status(500).json({ error: 'Failed to read file' })
    } else {
      return response.status(200).json(JSON.parse(data))
    }
  })
})

Táblázat generálás (1. típus)

Ez a kód egy HTML táblázatot hoz létre egy JavaScript tömb adatai alapján, majd hozzáadja azt a megfelelő div elemhez.

let div = document.getElementById('forTable')
let table = document.createElement('table')
let tbody = document.createElement('tbody')
for (const row of data) {
  let tr = document.createElement('tr')
  for (const item of Object.values(row)) {
    let td = document.createElement('td')
    td.textContent = item
    tr.appendChild(td)
  }
  tbody.appendChild(tr)
}
table.appendChild(tbody)
div.replaceChildren(table)

Táblázat generálás (2. típus)

Egy teljes HTML táblázatot generál egy üres div-be (`forTable`), beleértve a fejléceket és a tartalmat is.

const createTable = (array) => {
  const div = document.getElementById('forTable')
  const table = document.createElement('table')
  const thead = document.createElement('thead')
  const trForHeaders = document.createElement('tr')
  const headers = [`Név`, `Kor`]
  for (let i = 0; i < headers.length; i++) {
    const th = document.createElement('th')
    th.innerHTML = headers[i]
    trForHeaders.appendChild(th)
  }
  thead.appendChild(trForHeaders)
  table.appendChild(thead)
  const tbody = document.createElement('tbody')
  for (let i = 0; i < array.length; i++) {
    const trForContent = document.createElement('tr')
    let content = [array[i].nev, array[i].kor]
    for (let j = 0; j < content.length; j++) {
      let td = document.createElement('td')
      td.innerHTML = content[j]
      trForContent.appendChild(td)
    }
    tbody.appendChild(trForContent)
  }
  table.appendChild(tbody)
  div.appendChild(table)
}

Kattintott táblasor kijelölése

A `pilotaBody` azonosítójú táblázat törzs soraira kattintva csak egy sor jelölhető ki egyszerre. A kiválasztott sor háttérszínt kap és egyéni `data-selected` attribútummal jelöli a kijelölést.

const selectPilotaRows = () => {
  const select = (tbody) => {
    for (const tr of tbody.children) {
      tr.addEventListener('click', function () {
        const item = this
        console.log(item)
        if (item.dataset.selected === 'true') {
          item.dataset.selected = 'false'
          item.classList.remove('table-active')
        } else {
          for (const tr of tbody.children) {
            tr.dataset.selected = 'false'
            tr.classList.remove('table-active')
          }
          item.dataset.selected = 'true'
          item.classList.add('table-active')
        }
      })
    }
  }
  select(document.getElementById('pilotaBody'))
}

Összegzés (Elemi tétel)

Egy tömb elemeinek összegzésére szolgáló algoritmus hagyományos ciklussal.

function osszegzes(arr) {
  let sum = 0
  for (let i = 0; i < arr.length; i++) {
    sum += arr[i]
  }
  return sum
}

Megszámlálás (Elemi tétel)

Megszámolja, hány elem felel meg egy adott feltételnek egy tömbben.

function megszamlalas(arr, condition) {
  let count = 0
  for (let i = 0; i < arr.length; i++) {
    if (condition(arr[i])) {
      count++
    }
  }
  return count
}

function minErtek(arr) {
  let min = arr[0]

Szétválogatás (Elemi tétel)

A tömb elemeit két külön tömbbe osztja a feltétel alapján.

function szetvalogatas(arr, condition) {
  let megfelel = []
  let nemMegfelel = []
  for (let i = 0; i < arr.length; i++) {
    if (condition(arr[i])) {
      megfelel.push(arr[i])
    } else {
      nemMegfelel.push(arr[i])
    }
  }
  return [megfelel, nemMegfelel]
}

function novekvo(arr) {
  for (let i = 0; i < arr.length - 1; i++) {
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[i] > arr[j]) {
        let temp = arr[i]
        arr[i] = arr[j]
        arr[j] = temp
      }
    }
  }
  return arr
}
