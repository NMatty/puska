Fájlba írás (StreamWriter)

Ez a példa bemutatja, hogyan írjunk adatokat fájlba a C# alkalmazásban a StreamWriter használatával.

using System;
using System.Collections.Generic;
using System.IO;
using System.Text;
internal class Program
{
    /*Írás fájlba*/ public static void iras(List<Class> lista)
    {
        /*FileStream létrehozása (filenév, FileMode)*/ /*A Create létrehozza, vagy ha már létezik 
felülírja a fájlt*/ FileStream fs = new FileStream("file.txt", FileMode.Create);
        /*StreamWriter létrehozása (FileStream név, kódolás)*/ StreamWriter sw = new 
StreamWriter(fs, Encoding.UTF8);
        /*Fejléc írása, ha szükséges*/ sw.WriteLine("Header");
        /*Lista elemeinek írása*/ foreach (var item in lista)
        {
            /*Minden elem egy sorba írva*/ sw.WriteLine($"
            {
                item.Property1
            }
            ;
            {
                item.Property2
            }
            ");
        }
        sw.Close();
        /*Fájl lezárása*/ fs.Close();
        /*Fájlforrás lezárása*/
    }
    public static void feladat3(List<Class> lista)
    {
        Console.WriteLine($"3. feladat");
        Console.WriteLine($"A lista elemeinek száma:
        {
            lista.Count
        }
        ");
    }
    static void Main(string[] args)
    {
        /*1. feladat: létrehozunk egy listát*/ List<Class> adatok = new List<Class>
        {
            new Class("Adat1", "Value1"), /*Ez csak példa adat*/ new Class("Adat2", "Value2") /*Ez 
csak példa adat*/
        }
        ;
        /*2. feladat: fájlba írás*/ iras(adatok);
        /*3. feladat: lista kiírása*/ feladat3(adatok);
        Console.ReadKey();
    }
}
/*Példaként a Class osztályt feltételezve*/ public class Class
{
    public string Property1
    {
        get;
        set;
    }
    public string Property2
    {
        get;
        set;
    }
    public Class(string prop1, string prop2)
    {
        Property1 = prop1;
        Property2 = prop2;
    }
}

Fájlbeolvasás (StreamReader)

A `StreamReader` és `FileStream` segítségével történő fájlbeolvasás. A fájl sorait beolvassa és egy listába tölti, majd megszámolja az elemeket.

internal class Program
{
    public static List<Class> beolvas()
    {
        List<Class> lista = new List<Class>();
        FileStream fs = new FileStream("file.txt", FileMode.Open);
        StreamReader sr = new StreamReader(fs, Encoding.UTF8);
        sr.ReadLine();
        while (!sr.EndOfStream)
        {
            string sor = sr.ReadLine();
            lista.Add(new Class(sor.Split(';
            ')[0]));
        }
        sr.Close();
        fs.Close();
        return lista;
    }
    public static void feladat3(List<Class> lista)
    {
        Console.WriteLine($"3. feldat");
        Console.WriteLine($"A beolvasott adatsorok száma
        {
            lista.Count()
        }
        ");
    }
    static void Main(string[] args)
    {
        List<Class> adatok = beolvas();
        feladat3(adatok);
        Console.ReadKey();
    }
}

Szótár (dictionary)

A `Dictionary` használata C#-ban, ahol a kulcs egy `string`, az érték pedig egy `int`. A kód egy listából épít szótárat, ahol a kulcsok a stringek, az értékek pedig azok előfordulásainak száma.

public static void Dictionary(List<Class> lista)
{
    Dictionary<string, int> dictionary = new Dictionary<string, int>();
    for (int i = 0;
    i < lista.Count;
    i++)
    {
        if (dictionary.ContainsKey(lista[i].StringAdat))
        {
            dictionary[lista[i].StringAdat]++;
        }
        else
        {
            dictionary.Add(lista[i].StringAdat, 1);
        }
    }
    Console.WriteLine("Dictionary");
    foreach (KeyValuePair<string, int> kvp in dictionary)
    {
        Console.WriteLine($"\t
        {
            kvp.Key
        }
        -
        {
            kvp.Value
        }
        ");
    }
}

WPF több form kezelése (MainWindow és másik ablak kódja)

Ez a példa bemutatja, hogyan kezelhetünk több WPF ablakot (formot) egy alkalmazásban, például egy főablakot (MainWindow) és egy második ablakot (masikAblak). Az ablakok közötti navigáció és kommunikáció is bemutatásra kerül.

// MainWindow (egyik ablak) kódja:
public partial class MainWindow : Window
{
    public MainWindow()
    {
        InitializeComponent();
    }

    // Az alábbi függvény lefut, amikor a főoldal bezáródik, hogy az egész applikáció bezáruljon az 
összes ablakkal
    protected override void OnClosed(EventArgs e)
    {
        Application.Current.Shutdown();
    }

    // Függvény a másik oldal megnyitásához, például gombnyomásra:
    private void ugras(object sender, RoutedEventArgs e)
    {
        // Példány létrehozása, a másik oldalból, hogy meg tudjuk nyitni (valamint a jelenlegi 
ablak, "this" átadása, hogy később vissza tudjuk nyitni a jelenlegi oldalt, vagy kommunikáció 
céljából a két oldal között):
        masikAblak peldany = new masikAblak(this); // Ha esetleg egy listát vagy értéket kell 
használnunk a másik oldalon is, paraméterben a this mellet ezt is továbbadhatjuk.

        //Jelenlegi oldal elrejtése (A MainWindow-t csak rejtett/"Hide" lehet, mert ha bezárjuk, az 
alkalmazás is bezárul):
        this.Hide();

        // Másik ablak megjelenítése
        masikAblak.Show();
    }
}


// masikAblak kódja:
public partial class masikAblak: Window
{

    private MainWindow mw; // Létrehozunk egy példányt a MainWindow-ból

    // Konstruktor, amelynek segítségével értéket adunk a mw változónak. Itt a paraméterben kapott 
this érték jön át, és ha kell, itt inicializálható a továbbadott érték/lista is
    public masikAblak(MainWindow mw)
    {
        InitializeComponent();

        this.mw = mw;
    }

    // Itt is felülírjuk az OnClosed eseményt:
    protected override void OnClosed(EventArgs e)
    {
        this.Close(); // Jelenlegi ablak bezárása
        mw.Show(); // MainWindow megjelenítése
    }

    // Ezen az oldalon is egy esemény (pl gombnyomás) hatására, visszamegyünk a főoldalra:
    private void ugras(object sender, RoutedEventArgs e)
    {
        this.Close(); // Jelenlegi ablak bezárása
        mw.Show(); // MainWindow megjelenítése
    }
}

SQL Összekötés

Hozzunk létre egy osztályt a projekten belül, amely segítségével fel tudjuk dolgozni az adatbázis tartalmát.

class TesztOsztaly
{
    private int id;
    private string nev;
    private int kor;
    public TesztOsztaly(int id, string nev, int kor)
    {
        this.id = id;
        this.nev = nev;
        this.kor = kor;
    }
    public TesztOsztaly(string nev, int kor)
    {
        this.nev = nev;
        this.kor = kor;
    }
    public int Id
    {
        get => id;
        set => id = value;
    }
    public string Nev
    {
        get => nev;
        set => nev = value;
    }
    public int Kor
    {
        get => kor;
        set => kor = value;
    }
}

SQL Kapcsolat osztály

  1. Hozz létre egy osztályt, amely az SQL műveletek kezelésére szolgál.
  2. Deklarálj egy MySqlConnection típusú mezőt a kapcsolat létrehozásához.
  3. Hozz létre egy konstruktort, amely az alábbi paramétereket fogadja el:
    • host = SQL szerver cím
    • dbname = az adatbázis neve, amelyet kezelni szeretnél
    • ui = felhasználónév az adatbázishoz
    • pw = jelszó az adatbázishoz

class DBConnect {
    private MySqlConnection con;

    public DBConnect(string host, string dbname, string ui, string pw) {
        con = new MySqlConnection(
            $"Database = {dbname}; " +
            $"Data Source = {host}; " +
            $"User Id = {ui}; " +
            $"Password = {pw};"
        );
    }
}

SQL Adatok Beszúrása Adatbázisba

Ez a metódus beszúr egy új rekordot a teszttabla táblába. A kapcsolat megnyitásra kerül, ha sikeres, végrehajtja az INSERT lekérdezést. Paraméterezett lekérdezést használ a nev és kor értékek beszúrására. A beszúrás után a kapcsolat bezárásra kerül.

public void InsertData(TesztOsztaly temp)
{
    if(Connect())
    {
        string query = $"INSERT INTO teszttabla (nev, kor) VALUES (@nev, @kor);
        ";
        MySqlCommand cmd = new MySqlCommand(query, con);
        cmd.Parameters.AddWithValue("@nev", temp.Nev);
        cmd.Parameters.AddWithValue("@kor", temp.Kor);
        cmd.ExecuteNonQuery();
        ConnectClose();
    }
}

SQL Parancs, ahol több eredményt várunk (SELECT *)

Ez a metódus lekérdezi az összes rekordot a teszttabla táblából.
A kapcsolat megnyitása után egy SQL SELECT parancsot hajt végre.
A kapott eredményeket egy TesztOsztaly objektumokat tartalmazó listába menti.
A művelet végén bezárja az adatbázis kapcsolatot és visszaadja a listát.

public List<TesztOsztaly> SelectAll()
{
    List<TesztOsztaly> adatok = new List<TesztOsztaly>();
    if(Connect())
    {
        string query = $"SELECT * FROM teszttabla";
        MySqlCommand cmd = new MySqlCommand(query, con);
        MySqlDataReader reader = cmd.ExecuteReader();
        while(reader.Read())
        {
            adatok.Add(new TesztOsztaly(reader.GetInt32(0),reader.GetString(1), 
reader.GetInt32(2)));
        }
        ConnectClose();
    }
    return adatok;
}

SQL Parancs, ahol 1 eredményt várunk (AVG)

Ez a metódus lekérdezi a teszttabla táblából az átlagos életkort.
A kapcsolat megnyitása után egy SQL AVG() függvényt használó SELECT parancsot hajt végre.
Az eredményt egy double típusú változóba menti, ha van érvényes visszatérési érték.
A művelet végén bezárja az adatbázis kapcsolatot és visszaadja az átlagéletkort.

public double AtlagKor()
{
    double AvgAge = 0;
    if(Connect())
    {
        string query = $"SELECT AVG(kor) FROM teszttabla;
        ";
        MySqlCommand cmd = new MySqlCommand(query, con);
        if(cmd.ExecuteScalar() != null)
        {
            AvgAge = Convert.ToDouble(cmd.ExecuteScalar());
        }
        ConnectClose();
    }
    return AvgAge;
}

Származtatott osztály

Ez a példa bemutatja a származtatott osztályok használatát C#-ban. Az alap osztály (szülő) tartalmaz közös metódusokat, míg a származtatott osztályok (gyermekek) öröklik ezeket a metódusokat, és saját specifikus metódusokat adhatnak hozzá.

/* A származtatott osztályok */ /* Minden állat eszik és alszik, ezért csinálunk egy állat osztályt 
*/ class Animal  // Alaposztály (szülő)
{
    public void Eat()
    {
        Console.WriteLine("Az állat eszik.");
    }
    public void Sleep()
    {
        Console.WriteLine("Az állat alszik.");
    }
}
/* A kutya itt örökli az állatokra vonatkozó metódust, ezért nem kell kétszer leírni, sőt sajátot is
 adhatunk meg neki amivel csak Ő rendelkezik */ class Dog : Animal  // Származtatott osztály 
(gyermek)
{
    public void Bark()
    {
        Console.WriteLine("A kutya ugat: Vau vau!");
    }
}
/* A macskának pedig szintén megadjuk a saját metódusát, hiszen minden mást az állat osztályból 
örökölt */ class Cat : Animal  // Másik származtatott osztály
{
    public void Meow()
    {
        Console.WriteLine("A macska nyávog: Miaú!");
    }
}
class Program
{
    static void Main(string[] args)
    {
        Dog myDog = new Dog();
        myDog.Eat();
        // öröklött metódus myDog.Bark();
        // saját metódus Cat myCat = new Cat();
        myCat.Sleep();
        // öröklött metódus myCat.Meow();
        // saját metódus
    }
}

Abstraction

Ez a példa bemutatja az absztrakt osztályok és metódusok használatát C#-ban. Az absztrakt osztályok lehetővé teszik az osztályok öröklődését, miközben kötelezővé teszik a gyerekosztályok számára bizonyos metódusok megvalósítását.

using System;
/* Absztrakt osztály: Jármű */ abstract class Vehicle
{
    /* Absztrakt metódus: minden jármű máshogy indul el */ public abstract void Start();
    /* Normál metódus: minden jármű ugyanúgy áll meg */ public void Stop()
    {
        Console.WriteLine("A jármű megállt.");
    }
}
/* Leszármazott osztály: Autó */ class Car : Vehicle
{
    /* Az absztrakt metódus konkrét megvalósítása */ public override void Start()
    {
        Console.WriteLine("Az autó kulccsal indul.");
    }
}
/* Leszármazott osztály: Bicikli */ class Bicycle : Vehicle
{
    /* Az absztrakt metódus konkrét megvalósítása */ public override void Start()
    {
        Console.WriteLine("A bicikli pedálozással indul.");
    }
}
class Program
{
    static void Main(string[] args)
    {
        /* Példányosítunk egy autót, de Vehicle típusúként hivatkozunk rá */ Vehicle myCar = new 
Car();
        myCar.Start();
        /* "Az autó kulccsal indul." */ myCar.Stop();
        /* "A jármű megállt." */ /* Példányosítunk egy biciklit, szintén Vehicle típusúként */ 
Vehicle myBike = new Bicycle();
        myBike.Start();
        /* "A bicikli pedálozással indul." */ myBike.Stop();
        /* "A jármű megállt." */
    }
}
/* Az abstract class nem példányosítható közvetlenül, tehát nem lehet olyat csinálni, hogy new 
Vehicle(. Egy abstract metódusnak nincs törzse, a gyerekosztály kell megvalósítsa. Az absztrakt 
osztályban lehetnek normál metódusok is, amiket a gyerekek örökölnek. */

Math.Round(number); /*A legközelebbi egész számra kerekít*/
Math.Ceiling(number) /*Felfelé kerekít*/
Math.Floor(number) /*Lefelé kerekít*/
